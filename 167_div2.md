## Substring and Subsequence

<details>
<summary>Python</summary>

```python
def solve():
    a = input().strip()
    b = input().strip()
    ans = len(a)+len(b)
    for i in range(len(b)):
        ind = i
        for j in range(len(a)):
            if ind<len(b) and a[j] == b[ind]:
                ind+=1
        ans = min(ans,len(a)+len(b)+i-ind)

    print(ans)


for _ in range(int(input())):
    solve()

#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

void solve() {
    string a, b;
    cin >> a >> b;
    int ans = a.length() + b.length();
    for (int i = 0; i < b.length(); ++i) {
        int ind = i;
        for (int j = 0; j < a.length(); ++j) {
            if (ind < b.length() && a[j] == b[ind]) {
                ++ind;
            }
        }
        ans = min(ans, static_cast<int>(a.length() + b.length() + i - ind));
    }
    cout << ans << endl;
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}

```

import java.util.Scanner;

public class Main {
    public static void solve(Scanner sc) {
        String a = sc.next();
        String b = sc.next();
        int ans = a.length() + b.length();
        for (int i = 0; i < b.length(); ++i) {
            int ind = i;
            for (int j = 0; j < a.length(); ++j) {
                if (ind < b.length() && a.charAt(j) == b.charAt(ind)) {
                    ++ind;
                }
            }
            ans = Math.min(ans, a.length() + b.length() + i - ind);
        }
        System.out.println(ans);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int t = sc.nextInt();
        while (t-- > 0) {
            solve(sc);
        }
    }
}
