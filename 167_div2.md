## Substring and Subsequence

<summary>Python</summary>

```python
def solve():
    a = input().strip()
    b = input().strip()
    ans = len(a)+len(b)
    for i in range(len(b)):
        ind = i
        for j in range(len(a)):
            if ind<len(b) and a[j] == b[ind]:
                ind+=1
        ans = min(ans,len(a)+len(b)+i-ind)

    print(ans)


for _ in range(int(input())):
    solve()
```

```cpp
#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

void solve() {
    string a, b;
    cin >> a >> b;
    int ans = a.length() + b.length();
    for (int i = 0; i < b.length(); ++i) {
        int ind = i;
        for (int j = 0; j < a.length(); ++j) {
            if (ind < b.length() && a[j] == b[ind]) {
                ++ind;
            }
        }
        ans = min(ans, static_cast<int>(a.length() + b.length() + i - ind));
    }
    cout << ans << endl;
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```

```java
import java.util.Scanner;

public class Main {
    public static void solve(Scanner sc) {
        String a = sc.next();
        String b = sc.next();
        int ans = a.length() + b.length();
        for (int i = 0; i < b.length(); ++i) {
            int ind = i;
            for (int j = 0; j < a.length(); ++j) {
                if (ind < b.length() && a.charAt(j) == b.charAt(ind)) {
                    ++ind;
                }
            }
            ans = Math.min(ans, a.length() + b.length() + i - ind);
        }
        System.out.println(ans);
    }

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int t = sc.nextInt();
        while (t-- > 0) {
            solve(sc);
        }
    }
```

## Two Movies

<details>
<summary>Python</summary>

```python
def solve():
    n = int(input())
    a = list(map(int,input().split()))
    b = list(map(int,input().split()))
    ra,rb = 0,0
    pos,neg = 0,0
    for i in range(n):
        if a[i] == 1 and b[i] == 1:
            pos+=1
        elif a[i] == -1 and b[i] == -1:
            neg+=1
        elif a[i]>b[i]:
            ra+=a[i]
        else:
            rb+=b[i]
    while pos>0:
        if ra>rb:
            rb+=1
        else:
            ra+=1
        pos-=1
    while neg>0:
        if ra>rb:
            ra-=1
        else:
            rb-=1
        neg-=1
    print(min(ra,rb))




for _ in range(int(input())):
    solve()
```

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

void solve() {
    int n;
    cin >> n;
    vector<int> a(n);
    vector<int> b(n);
    
    for (int i = 0; i < n; ++i) {
        cin >> a[i];
    }
    for (int i = 0; i < n; ++i) {
        cin >> b[i];
    }
    
    int ra = 0, rb = 0;
    int pos = 0, neg = 0;
    
    for (int i = 0; i < n; ++i) {
        if (a[i] == 1 && b[i] == 1) {
            pos++;
        } else if (a[i] == -1 && b[i] == -1) {
            neg++;
        } else if (a[i] > b[i]) {
            ra += a[i];
        } else {
            rb += b[i];
        }
    }
    
    while (pos > 0) {
        if (ra > rb) {
            rb++;
        } else {
            ra++;
        }
        pos--;
    }
    
    while (neg > 0) {
        if (ra > rb) {
            ra--;
        } else {
            rb--;
        }
        neg--;
    }
    
    cout << min(ra, rb) << endl;
}

int main() {
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
```
