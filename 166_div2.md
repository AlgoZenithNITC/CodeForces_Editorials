        if s[i].isdigit():

    i = 0

            if i + 1 < n:

import java.util.Scanner;

public class Main {
    public static boolean solve(String s) {
        int n = s.length();
        int i = 0;
        int numBefore = 0;
        while (i < n) {
            if (Character.isDigit(s.charAt(i))) {
                if (s.charAt(i) - '0' >= numBefore) {
                    numBefore = s.charAt(i) - '0';
                } else {
                    return false;
                }
            }
            i++;
        }
        i = 0;
        char letterBefore = 'a';
        while (i < n) {
            if (Character.isDigit(s.charAt(i))) {
                i++;
                continue;
            }
            if (s.charAt(i) < 97) {
                return false;
            }
            if (Character.isLetter(s.charAt(i))) {
                if (s.charAt(i) >= letterBefore) {
                    letterBefore = s.charAt(i);
                } else {
                    return false;
                }
            }
            i++;
        }
        i = 0;
        while (i < n) {
            if (Character.isDigit(s.charAt(i))) {
                i++;
                continue;
            }
            if (Character.isLetter(s.charAt(i))) {
                if (i + 1 < n) {
                    if (Character.isDigit(s.charAt(i + 1))) {
                        return false;
                    }
                }
            }
            i++;
        }
        return true;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int t = scanner.nextInt();
        scanner.nextLine();
        while (t-- > 0) {
            int n = scanner.nextInt();
            scanner.nextLine();
            String s = scanner.nextLine();
            boolean ans = solve(s);
            System.out.println(ans ? "YES" : "NO");
        }
    }
}

t = int(input())
for _ in range(t):
    n, m = map(int, input().split())
    p = list(map(int, input().split()))
    t_list = list(map(int, input().split()))
    prog = []
    test = []
    for i in range(n + m + 1):
        a = (p[i], i)
        b = (t_list[i], i)
        if len(prog) == n + 1:
            test.append(b)
            continue
        if len(test) == m + 1:
            prog.append(a)
            continue
        if a[0] > b[0]:
            prog.append(a)
        else:
            test.append(b)
    sum_val = sum(p) + sum(t_list)
    ans = [sum_val] * (n + m + 1)
    if len(prog) == n + 1:
        for i in prog:
            ans[i[1]] -= i[0]
        last = prog[-1][1]
        res = t_list[last] - p[last]
        for i in test:
            ans[i[1]] += res - i[0]
    else:
        for i in test:
            ans[i[1]] -= i[0]
        last = test[-1][1]
        res = p[last] - t_list[last]
        for i in prog:
            ans[i[1]] += res - i[0]
    print(*ans)

import java.util.Scanner;
import java.util.ArrayList;
import java.util.List;
import java.util.Collections;

public class Main {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int t = scanner.nextInt();
        while (t-- > 0) {
            int n = scanner.nextInt();
            int m = scanner.nextInt();
            List<Integer> p = new ArrayList<>();
            List<Integer> tList = new ArrayList<>();
            for (int i = 0; i < n + m + 1; i++) p.add(scanner.nextInt());
            for (int i = 0; i < n + m + 1; i++) tList.add(scanner.nextInt());
            List<Pair<Integer, Integer>> prog = new ArrayList<>();
            List<Pair<Integer, Integer>> test = new ArrayList<>();
            for (int i = 0; i < n + m + 1; i++) {
                Pair<Integer, Integer> a = new Pair<>(p.get(i), i);
                Pair<Integer, Integer> b = new Pair<>(tList.get(i), i);
                if (prog.size() == n + 1) {
                    test.add(b);
                    continue;
                }
                if (test.size() == m + 1) {
                    prog.add(a);
                    continue;
                }
                if (a.getFirst() > b.getFirst()) {
                    prog.add(a);
                } else {
                    test.add(b);
                }
            }
            int sum = 0;
            for (Pair<Integer, Integer> i : prog) sum += i.getFirst();
            for (Pair<Integer, Integer> i : test) sum += i.getFirst();
            List<Integer> ans = new ArrayList<>(Collections.nCopies(n + m + 1, sum));
            if (prog.size() == n + 1) {
                for (Pair<Integer, Integer> i : prog) {
                    ans.set(i.getSecond(), ans.get(i.getSecond()) - i.getFirst());
                }
                int last = prog.get(prog.size() - 1).getSecond();
                int res = tList.get(last) - p.get(last);
                for (Pair<Integer, Integer> i : test) {
                    ans.set(i.getSecond(), ans.get(i.getSecond()) + res - i.getFirst());
                }
            } else {
                for (Pair<Integer, Integer> i : test) {
                    ans.set(i.getSecond(), ans.get(i.getSecond()) - i.getFirst());
                }
                int last = test.get(test.size() - 1).getSecond();
                int res = p.get(last) - tList.get(last);
                for (Pair<Integer, Integer> i : prog) {
                    ans.set(i.getSecond(), ans.get(i.getSecond()) + res - i.getFirst());
                }
            }
            for (int i : ans) System.out.print(i + " ");
            System.out.println();
        }
    }

    static class Pair<F, S> {
        private F first;
        private S second;

        public Pair(F first, S second) {
            this.first = first;
            this.second = second;
        }

        public F getFirst() {
            return first;
        }

        public S getSecond() {
            return second;
        }
    }
}
