## Have Your Cake and Eat It Too

<details>
<summary>Python</summary>

```python
import sys
import itertools

def main():
    input = sys.stdin.read
    data = input().split()
    idx = 0

    t = int(data[idx])
    idx += 1

    results = []
    for _ in range(t):
        n = int(data[idx])
        idx += 1

        val = [[0] * (n + 1) for _ in range(3)]
        pf = [[0] * (n + 1) for _ in range(3)]

        for i in range(3):
            for j in range(1, n + 1):
                val[i][j] = int(data[idx])
                idx += 1
                pf[i][j] = pf[i][j - 1] + val[i][j]

        ok = False
        perm = [0, 1, 2]
        comp = (pf[perm[0]][n] + 2) // 3

        for perm in itertools.permutations([0, 1, 2]):
            cur = 1
            while cur <= n and pf[perm[0]][cur] < comp:
                cur += 1
            for j in range(cur + 1, n):
                if pf[perm[1]][j] - pf[perm[1]][cur] >= comp and pf[perm[2]][n] - pf[perm[2]][j] >= comp:
                    ans = [None] * 3
                    ans[perm[0]] = (1, cur)
                    ans[perm[1]] = (cur + 1, j)
                    ans[perm[2]] = (j + 1, n)
                    results.append(' '.join(f'{x[0]} {x[1]}' for x in ans))
                    ok = True
                    break
            if ok:
                break

        if not ok:
            results.append('-1')

    print('\n'.join(results))

if __name__ == "__main__":
    main()
```

## Swap Dilemma

<details>
<summary>Python</summary>

```python
def inversions(arr, l, r):
    if r == l:
        return 0

    # Divide and conquer:
    mid = (l + r) // 2
    x = inversions(arr, l, mid)
    y = inversions(arr, mid + 1, r)

    # Simple merging:
    ans = [0] * (r - l + 1)

    curr = 0
    inv = 0
    pointx = l
    pointy = mid + 1

    while pointx <= mid and pointy <= r:
        if arr[pointx] < arr[pointy]:
            inv += pointy - mid - 1
            ans[curr] = arr[pointx]
            pointx += 1
        else:
            ans[curr] = arr[pointy]
            pointy += 1
        curr += 1

    while pointx <= mid:
        inv += pointy - mid - 1
        ans[curr] = arr[pointx]
        pointx += 1
        curr += 1

    while pointy <= r:
        ans[curr] = arr[pointy]
        pointy += 1
        curr += 1

    # Final writeback:
    for i in range(l, r + 1):
        arr[i] = ans[i - l]

    return x + y + inv
```
