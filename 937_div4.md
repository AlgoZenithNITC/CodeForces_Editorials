## Stair, Peak, or Neither?

<details>
<summary>Python</summary>

```python
t = int(input())
while t > 0:
    a, b, c = map(int, input().split())
    if a < b and b < c:
        print("STAIR")
    elif a < b and b > c:
        print("PEAK")
    else:
        print("NONE")
    t -= 1
```

## Product of Binary Decimals

<details>
<summary>Python</summary>

```python
def generate(t, n, v):
    if n == 0:
        k = int(t)
        if k <= 1e5:
            v.append(k)
        return
    generate(t + '0', n - 1, v)
    generate(t + '1', n - 1, v)


def solve(n, m):
    if n == 1:
        return True
    flag = False
    for i in m.keys():
        if n % i == 0:
            flag |= solve(n // i, m)
            if flag:
                return True
    return flag


if __name__ == "__main__":
    t = int(input())
    v = []
    generate("", 6, v)
    m = {i: 1 for i in v if i > 1}
    for _ in range(t):
        n = int(input())
        if n in m:
            print("YES")
            continue
        flag = solve(n, m)
        print("YES" if flag else "NO")
```

```java
import java.util.*;

public class Main {

    static void generate(String t, int n, ArrayList<Integer> v) {
        if (n == 0) {
            int k = Integer.parseInt(t);
            if (k <= 1e5) v.add(k);
            return;
        }
        generate(t + "0", n - 1, v);
        generate(t + "1", n - 1, v);
    }

    static boolean solve(int n, HashMap<Integer, Integer> m) {
        if (n == 1) return true;
        boolean flag = false;
        for (Map.Entry<Integer, Integer> entry : m.entrySet()) {
            int key = entry.getKey();
            if (n % key == 0) {
                flag |= solve(n / key, m);
                if (flag) return true;
            }
        }
        return flag;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int t = scanner.nextInt();
        ArrayList<Integer> v = new ArrayList<>();
        generate("", 6, v);
        HashMap<Integer, Integer> m = new HashMap<>();
        for (int i : v) {
            if (i > 1) m.put(i, 1);
        }
        while (t-- > 0) {
            int n = scanner.nextInt();
            if (m.containsKey(n)) {
                System.out.println("YES");
                continue;
            }
            boolean flag = solve(n, m);
            System.out.println(flag ? "YES" : "NO");
        }
    }
}
```

## Nearly Shortest Repeating Substring

<details>
<summary>Python</summary>

```python
def solve(a, b):
    c = 0
    for i in range(len(a)):
        if a[i] != b[i]:
            c += 1
    if c > 1:
        return False
    return True

if __name__ == "__main__":
    t = int(input())
    for _ in range(t):
        n = int(input())
        s = input()
        flag = False
        for i in range(1, n + 1):
            if n % i == 0:
                t1 = s[:i]
                t2 = s[n - i:]
                s1, s2 = "", ""
                for j in range(1, n // i + 1):
                    s1 += t1
                    s2 += t2
                if solve(s1, s) or solve(s2, s):
                    print(i)
                    flag = True
                    break
        if flag:
            break
```

```java
import java.util.Scanner;

public class Main {
    public static boolean solve(String a, String b) {
        int c = 0;
        for (int i = 0; i < a.length(); i++) {
            if (a.charAt(i) != b.charAt(i)) {
                c++;
            }
        }
        return c <= 1;
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int t = scanner.nextInt();
        scanner.nextLine(); // Consume newline character
        while (t-- > 0) {
            int n = scanner.nextInt();
            scanner.nextLine(); // Consume newline character
            String s = scanner.nextLine();
            boolean flag = false;
            for (int i = 1; i <= n; i++) {
                if (n % i == 0) {
                    String t1 = s.substring(0, i);
                    String t2 = s.substring(n - i);
                    StringBuilder s1 = new StringBuilder();
                    StringBuilder s2 = new StringBuilder();
                    for (int j = 0; j < n / i; j++) {
                        s1.append(t1);
                        s2.append(t2);
                    }
                    if (solve(s1.toString(), s) || solve(s2.toString(), s)) {
                        System.out.println(i);
                        flag = true;
                        break;
                    }
                }
            }
            if (flag) {
                break;
            }
        }
    }
}
```
