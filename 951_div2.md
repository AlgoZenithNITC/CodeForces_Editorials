## Guess the Maximum

<details>
<summary>Python</summary>

```python
def main():
    t = int(input())
    for _ in range(t):
        n = int(input())
        a = list(map(int, input().split()))
        mini = max(a[0], a[1])
        for i in range(1, n - 1):
            mini = min(mini, max(a[i], a[i + 1]))
        print(mini - 1)

if __name__ == "__main__":
    main()
```

## XOR Sequences

<details>
<summary>Python</summary>

```python
def main():
    import sys
    input = sys.stdin.read
    data = input().split()
    
    t = int(data[0])
    index = 1
    results = []
    
    for _ in range(t):
        a = int(data[index])
        b = int(data[index + 1])
        index += 2
        
        for i in range(30):
            if (a & (1 << i)) != (b & (1 << i)):
                results.append(1 << i)
                break
    
    for result in results:
        print(result)

if __name__ == "__main__":
    main()
```

## Earning on Bets

```python
from math import gcd
from functools import reduce

def lcm(a, b):
    return a * b // gcd(a, b)

def solve():
    n = int(input())
    k = list(map(int, input().split()))
```

```cpp
#include <bits/stdc++.h>
using namespace std;

#define int long long

int gcd(int a, int b) {
    while (b != 0) {
        int tmp = a % b;
        a = b;
        b = tmp;
    }
    return a;
}
int lcm(int a, int b) {
    return a * b / gcd(a, b);
}

void solve() {
    int n;
    cin >> n;
    
    vector <int> k(n);
    for (int i = 0; i < n; i++) {
        cin >> k[i];
    }
    
    int z = 1;
    for (int i = 0; i < n; i++) {
        z = lcm(z, k[i]);
    }
    
    int suma = 0;
    for (int i = 0; i < n; i++) {
        suma += z / k[i];
    }
    
    if (suma < z) {
        for (int i = 0; i < n; i++) {
            cout << z / k[i] << " ";
        }
        cout << "\n";
    } else {
        cout << -1 << "\n";
    }
}

signed main() {
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
}
```

```java
import java.util.Scanner;
import java.util.ArrayList;

public class Main {
    
    // Method to compute the Greatest Common Divisor (GCD)
    public static long gcd(long a, long b) {
        while (b != 0) {
            long tmp = a % b;
            a = b;
            b = tmp;
        }
        return a;
    }

    // Method to compute the Least Common Multiple (LCM)
    public static long lcm(long a, long b) {
        return a * b / gcd(a, b);
    }

    // Method to solve the problem for one test case
    public static void solve(Scanner scanner) {
        int n = scanner.nextInt();
        
        long[] k = new long[n];
        for (int i = 0; i < n; i++) {
            k[i] = scanner.nextLong();
        }
        
        long z = 1;
        for (int i = 0; i < n; i++) {
            z = lcm(z, k[i]);
        }
        
        long suma = 0;
        for (int i = 0; i < n; i++) {
            suma += z / k[i];
        }
        
        if (suma < z) {
            for (int i = 0; i < n; i++) {
                System.out.print(z / k[i] + " ");
            }
            System.out.println();
        } else {
            System.out.println(-1);
        }
    }

    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        int t = scanner.nextInt();
        while (t-- > 0) {
            solve(scanner);
        }
        scanner.close();
    }
}
```
